Demo2 - Dynamic Table

    my_list is an array within angular scope ($scope.my_list)
    The table is populated with the array data in my_list with ng-repeat feature of angular
Header1 Header2 Header3
{{ x.field1 }} {{ x.field2 }} More Details

Dynamic row details



{{key}} : {{ value }}


HTML
      
 <div ng-app="myApp" ng-controller="myCtrl">    
  	<table id="my_table" class="table table-chrome table-striped" style="display: none;table-layout: fixed;">
		  <thead class="bulk-table-head">
			<th>Header1</th>
			<th>Header2</th>
            <th>Header3</th>
		  </thead>
		  <tbody class="pages-listings">
			<tr ng-repeat="x in my_list">
				<td>{{ x.field1 }}</td>
				<td>{{ x.field2 }}</td>
                <td><a ng-click="action(x)">More Details</a></td>
			</tr>
		  </tbody>
		</table>
        <div id="details">
      		<h3>Dynamic row details</h3><BR/>
          <BR/>  
          <table>
  			<tr ng-repeat="(key, value) in row_details">
    		<td> {{key}} :</td> <td> {{ value }} </td>
  			</tr>
		</table>
      	</div>
  </div>

JS
      
// required app for angular
var app = angular.module('myApp', []);

app.controller('myCtrl',['$scope',function($scope) {
	require(["splunkjs/mvc",'splunkjs/mvc/simplexml/ready!'],function(mvc) {
		$scope.my_list = [];
		$scope.my_list.push({'field1':'data1_A','field2':'data1_B','field3':'data1_C','field4':'data1_D','field5':'data1_E','field6':'data1_F'});
		$scope.my_list.push({'field1':'data2_A','field2':'data2_B','field3':'data2_C','field4':'data2_D','field5':'data2_E','field6':'data2_F'});
		$scope.my_list.push({'field1':'data3_A','field2':'data3_B','field3':'data3_C','field4':'data3_D','field5':'data3_E','field6':'data3_F'});
		console.log($scope.my_list);
		
		$scope.action = function(row){
			console.log(row);
			$scope.row_details = row;
		}
		
		$scope.$apply();
		$('#my_table').show();
	});
}]);
